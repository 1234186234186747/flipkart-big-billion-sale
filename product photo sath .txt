        function openModal(p) {
            document.getElementById('mName').innerText = p.name;
            document.getElementById('mPrice').innerText = "₹" + p.price.toLocaleString();
            document.getElementById('finalP').innerText = "₹" + p.price.toLocaleString();
            document.getElementById('mImg').src = p.img;
            activeAmt = p.price;

            const colorOpts = document.getElementById('colorOptions');
            const colorNameText = document.getElementById('colorName');
            
            if(p.variants && p.variants.length > 1) {
                document.getElementById('colorSection').style.display = 'block';
                colorOpts.innerHTML = p.variants.map((v, index) => `
                    <div class="color-dot ${index === 0 ? 'active' : ''}" 
                         style="background-color: ${v.hex}" 
                         onclick="selectColor('${v.img}', '${v.color}', this)">
                    </div>
                `).join('');
                colorNameText.innerText = p.variants[0].color;
            } else {
                document.getElementById('colorSection').style.display = 'none';
            }

            document.getElementById('pModal').classList.remove('hidden');
            document.body.style.overflow = 'hidden';
        }

        function selectColor(imgUrl, name, element) {
            document.getElementById('mImg').src = imgUrl;
            document.getElementById('colorName').innerText = name;
            document.querySelectorAll('.color-dot').forEach(dot => dot.classList.remove('active'));
            element.classList.add('active');
        }

        function closeModal() {
            document.getElementById('pModal').classList.add('hidden');
            document.body.style.overflow = 'auto';
        }
